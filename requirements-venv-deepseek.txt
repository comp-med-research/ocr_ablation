#   cd ocr_ablation
#   python3 -m venv .venv-deepseek
#   source .venv-deepseek/bin/activate    # Windows: .venv-deepseek\Scripts\activate
#   pip install -U pip
#
# Install PyTorch **first** for your CUDA (pick the command from https://pytorch.org/ ).
#
# Blackwell (e.g. RTX PRO 6000, sm_120): if you see
#   "CUDA error: no kernel image is available for execution on the device"
# your PyTorch wheel was built without kernels for sm_120. Install a Blackwell-capable build from:
#   https://pytorch.org/get-started/locally/
# Example (verify URL on that page — nightly moves):
#   pip install --pre torch torchvision torchaudio --index-url https://download.pytorch.org/whl/nightly/cu128
#
# Temporary CPU-only workaround (very slow):
#   export DEEPSEEK_DEVICE=cpu
#
# Example for older GPUs (CUDA 12.4):
#   pip install torch torchvision torchaudio --index-url https://download.pytorch.org/whl/cu124
#
# Then:
#   pip install -r requirements-venv-deepseek.txt
#
# Optional — faster attention if it builds on your GPU (often finicky on brand-new SMs):
#   pip install flash-attn==2.7.3 --no-build-isolation
# The loader skips the ``flash_attention_2`` attempt when ``flash_attn`` is not installed (avoids a
# misleading “install flash_attn” message when the real failure was an earlier impl).
